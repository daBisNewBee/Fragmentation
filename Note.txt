start: 启动Fragment(启动者和被启动者是在同一个栈的)
start(CycleFragment.newInstance(mNumber + 1));

// 以某种启动模式，启动新的Fragment
start(SupportFragment fragment, int launchMode)

startWithPop: 启动Fragment，并关闭自己
startWithPop(CycleFragment.newInstance(mNumber + 1));

startForResult: 启动Fragment，并携带返回值
startForResult(ModifyDetailFragment.newInstance(mTitle), REQ_MODIFY_FRAGMENT);
